# Secret Detection Patterns
# Each line is a regex pattern (comments start with #)

# AWS Keys
AKIA[0-9A-Z]{16}
aws_access_key_id\s*[:=]\s*['"][A-Z0-9]{20}['"]
aws_secret_access_key\s*[:=]\s*['"][A-Za-z0-9/+=]{40}['"]

# GitHub Tokens
ghp_[a-zA-Z0-9]{36}
gho_[a-zA-Z0-9]{36}
ghu_[a-zA-Z0-9]{36}
ghs_[a-zA-Z0-9]{36}
ghr_[a-zA-Z0-9]{36}
github_pat_[a-zA-Z0-9]{22}_[a-zA-Z0-9]{59}

# OpenAI Keys
sk-[a-zA-Z0-9]{48}
sk-proj-[a-zA-Z0-9]{48}

# Anthropic Keys
sk-ant-[a-zA-Z0-9-]{90,}

# Slack Tokens
xox[baprs]-[0-9]{10,13}-[0-9]{10,13}-[a-zA-Z0-9]{24}

# Google API Keys
AIza[0-9A-Za-z_-]{35}

# Private Keys
-----BEGIN (RSA|DSA|EC|OPENSSH|PGP) PRIVATE KEY-----
-----BEGIN PRIVATE KEY-----
-----BEGIN ENCRYPTED PRIVATE KEY-----

# Generic Secrets (quoted values with common key names)
password\s*[:=]\s*['"][^'"]{8,}['"]
secret\s*[:=]\s*['"][^'"]{8,}['"]
api_key\s*[:=]\s*['"][^'"]{8,}['"]
apikey\s*[:=]\s*['"][^'"]{8,}['"]
access_token\s*[:=]\s*['"][^'"]{8,}['"]
auth_token\s*[:=]\s*['"][^'"]{8,}['"]

# JWT Tokens (only match actual tokens, not patterns)
eyJ[a-zA-Z0-9_-]{10,}\.eyJ[a-zA-Z0-9_-]{10,}\.[a-zA-Z0-9_-]{10,}
